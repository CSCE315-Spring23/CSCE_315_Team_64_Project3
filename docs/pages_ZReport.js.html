/**
 * Renders a Z Report table that displays data fetched from an external API.
 * @function
 * @returns {JSX.Element} - The rendered Z Report table.
 */
function ZReport() {
  const [data, setZReport] = useState([]);
  
  const populate = async() => {
    try {
      const response = await fetch("http://localhost:8000/zrepfull")
      const jsonData = await response.json()
      setZReport(jsonData)
      console.log(jsonData)
    } catch (err) {
      console.log(err.message)
    }
  }
  
  useEffect(() => {
    populate();
  },[]);
  
  return (
    <div className="XReport">
        <NavbarComp/>
        <h1 class="inv">Z Report</h1>
      <table>
        <tr>
          <th>Item</th>
          <th>Name</th>
          <th>Price</th>
        </tr>
        {data.map((val, key) => {
          return (
            <tr key={key}>
              <td>{val.zrep_id}</td>
              <td>{val.zrep_items}</td>
              <td>{val.zrep_price}</td>
            </tr>
          )
        })}
      </table>
    </div>
  );
}
  
export default ZReport;
